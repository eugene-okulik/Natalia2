FROM students s
left join books b
on s.id = b.taken_by_student_id
left join `groups` g
on s.group_id = g.id
left join marks m
on s.id = m.student_id
left join lessons l
on m.lesson_id = l.id
left join subjects w
on l.subject_id = w.id
WHERE s.id = 22416
